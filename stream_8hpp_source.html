133  {
134  unsigned char id_ = 1;
135  explicit token(unsigned char id) : id_(id) {}
136  public:
137  token() = default;
138  token(token const&) = default;
139  operator bool() const { return id_ != 0; }
140  bool operator==(token const& t) { return id_ == t.id_; }
141  bool operator!=(token const& t) { return id_ != t.id_; }

143  void reset() { id_ = 0; }
